1004. Max Consecutive Ones III
Medium

class Solution {
    public int longestOnes(int[] nums, int k) {
        int zeroCount=0,i=0;
        int ans=0;
        for(int j=0;j<nums.length;j++){
            if(nums[j]==0)  zeroCount++;
            while(zeroCount>k){
                if(nums[i]==0)  zeroCount--;
                i++;
            }
            ans = Math.max(ans,j-i+1);
        }
        return ans;
    }
}
